\noindent
Read all the problems before starting to solve any! 

\begin{prb}
Prove that if $b$ is any real number and $a \neq 0$, then the quadratic equation $ax^2 + bx - a = 0$ has two distinct real roots.
\end{prb}


\begin{prb}
$f(x) = x^2 + px + q$ is an even function. Find (with proof) the value of $p$. 
\end{prb}


\begin{prb}
Points $A, B, C$  lie on a circle of radius $r$, and $AC = r$. Find $\angle ABC$. \\
\textit{Hint: Is there only one solution?}
\end{prb}


\begin{prb}
	\part{a} Find $p$ and $q$, if $10$ and $-15$ are the roots of the quadratic equation $x^2 + px + q$.
	\part{b} Sum of squares of roots of a quadratic equation $x^2 + px - 3$ is equal to $10$. Find $p$.
\end{prb}


\begin{prb}
	\part{a} What is the minimum value of the function $f(x) = x^2 + 10x - 13$? 
	\part{b} What is its maximum value?
\end{prb}




\begin{prb}
Let $O$ be the center of a circle and $A$ be a point outside of it. Let $AM$ and $AN$ be tangents to the circle, 
and let $B$ be the intersection of line $AO$ with the circle furthest from $A$. Given that $AN \parallel BM$, find $\angle OAM$ and prove that $AMBN$ is a rhombus.
\end{prb}
