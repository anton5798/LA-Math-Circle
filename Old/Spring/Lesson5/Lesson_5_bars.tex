\begin{prb}
\part{a} Show that $$1+2{n \choose 1} + 2^2{n \choose 2} + \ldots + 2^n{n \choose n} = 3^n$$
\part{b} Show that 
$$2^{2n}{{2n} \choose {2n}} - 2^{2n-1}{{2n} \choose {2n-1}} + \ldots + 2^{2}{{2n} \choose {2}} - 2{{2n} \choose {1}} = 0$$

\end{prb}

\begin{prb}
Show that the number of ways to write a positive integer $n$ as a sum of $k$ nonnegative numbers is $${{n+k-1} \choose {k-1}}$$
\end{prb}

\begin{prb}
At a math circle meeting 10 students are solving 10 problems. Any two students solved a different number of problems, and every problem is solved by the same number of students. Viserion solved problems 1 through 5 and did not solve problems 6 through 9. Did he solve problem 10?
\end{prb}

\begin{prb}
A toy consists of a ring with 3 red beads and 7 blue beads on it. If two configurations of beads differ only by rotations and reflections, they are considered the same toy. How many different toys are there?
\end{prb}

\begin{prb}
Let $K$ be a point on the side $CD$ of a square $ABCD$ such that $CK/KD = 1/2$. If the side length of the square is 1, find the length of the perpendicular from $C$ to the line $AK$. You may use the Pythagorean theorem.
\end{prb}

\begin{prb}
Let $AA_1$, $BB_1$ and $CC_1$ be the altitudes of the acute triangle $\triangle ABC$. Show that if $A_1B_1 \parallel AB$ and $B_1C_1 \parallel BC$, then $A_1C_1 \parallel AC$.
\end{prb}
