\section{From Last Week}

\begin{cprb}{2}[Konstantin's group only]
A function $f : \R \to \R$ is called \textit{even} if $f(x) = f(-x)$ for all $x \in \R$. Similarly, a function is called \textit{odd} if $f(x)= -f(-x)$ for all $x$.\\
\part{b} Show that any function from $\R$ to $\R$ can be uniquely written as a sum of an even and an odd function.
\end{cprb}

\begin{cprb}{3}
Find all functions $f : \R \to \R$ such that $f(2x+1) = 4x^2+14x+7$.
\end{cprb}

\begin{cprb}{4}
Five integers are written on the board -- three coefficients of a quadratic equation and two roots in arbitrary order. After one of the numbers is erased, the numbers $2,3,4,-5$ are left. What number was erased?
\end{cprb}

\begin{cprb}{5}[Anton's group only]
Let $ABCD$ be a quadrilateral such that there exists a circle tangent to all of its four sides. Such a quadrilateral is called \textit{circumscribed}. Show that $AB + CD = AD + BC$.
\end{cprb}


\section{New Problems}

\begin{cprb}{1}
Let $f(x) = ax^2+bx+c$ be a quadratic equation with $a > 0$.\\ 
\part{a} Show that if $f$ has no real roots, then $f(x) > 0$ for all real $x$.\\
\textit{Hint: complete the square!}
\part{b} Show that if $f$ has exactly one real root $x_0$, then $f(x) > 0$ for all real $x \neq x_0$.
\part{c} Show that if $f$ has exactly two real roots $x_0 < x_1$, then $f(x) < 0$ for all real $x_0 < x < x_1$ and $f(x) > 0$ for all $x > x_1$ and $x < x_0$. 
\part{d} Formulate and prove the analogues of parts a), b), c) for the case when $a < 0$.\\
\end{cprb}

\begin{cprb}{2}
Let $f(x) = ax^2+bx+c$ be a quadratic equation with $a > 0$.\\ 
Show that $f$ achieves its unique minimal value at $-b/(2a)$. In other words, show that for any $x \neq -b/(2a)$ we have $$f(x) > f \left ( \frac{-b}{2a} \right) $$ Show that if $a < 0$, then similarly $f$ achieves its unique maximal value at $-b/(2a)$.
\end{cprb}

\begin{cprb}{3}
Find all solutions to the equation $x(x+1) = 2018\cdot2019$
\end{cprb}
